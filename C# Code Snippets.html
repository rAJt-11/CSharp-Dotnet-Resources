5. Find the first non-repeated character

string input5 = "swiss";
char rslt = input5.GroupBy(c => c)
       .Where(g => g.Count() == 1)
       .Select(g => g.Key)
       .FirstOrDefault();
Console.WriteLine("First non-repeated character - " + rslt);
First non-repeated character - w

6. Find the most frequent character

string input6 = "test";
char mostFrequent = input6.GroupBy(c => c)
                 .OrderByDescending(g => g.Count())
                 .First().Key;
Console.WriteLine("Most frequent character - " + mostFrequent);
Most frequent character - t

Is anagrams - True

8. Count character frequencies

string input8 = "aabbcc";
var groups = input8.GroupBy(c => c)
          .ToDictionary(g => g.Key, g => g.Count());
foreach (var pair in groups)
    Console.WriteLine($"{pair.Key} appears {pair.Value} times.");
a appears 2 times.
b appears 2 times.
c appears 2 times.

11. Find the longest word in a sentence

13. Convert string to title case

Title Case - Hello World
